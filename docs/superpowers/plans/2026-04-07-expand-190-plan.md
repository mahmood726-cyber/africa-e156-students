# lib/stats_library.py
"""
Pure-Python statistical methods library for Africa E156 papers.
No external dependencies beyond math/random. Student-portable.
"""
import math
import random

def bootstrap_ci(data, stat_fn=None, n_boot=10000, alpha=0.05, seed=42):
    """BCa bootstrap confidence interval."""
    rng = random.Random(seed)
    if stat_fn is None:
        stat_fn = lambda x: sum(x) / len(x) if x else 0
    observed = stat_fn(data)
    boot_stats = []
    n = len(data)
    for _ in range(n_boot):
        sample = [data[rng.randint(0, n-1)] for _ in range(n)]
        boot_stats.append(stat_fn(sample))
    boot_stats.sort()
    lo = boot_stats[int(alpha/2 * n_boot)]
    hi = boot_stats[int((1-alpha/2) * n_boot)]
    return {"estimate": observed, "ci_lower": lo, "ci_upper": hi, "n_boot": n_boot}

def gini_coefficient(values):
    """Gini coefficient for inequality measurement."""
    if not values or all(v == 0 for v in values):
        return 0.0
    sorted_vals = sorted(values)
    n = len(sorted_vals)
    total = sum(sorted_vals)
    if total == 0:
        return 0.0
    cumsum = 0
    gini_sum = 0
    for i, v in enumerate(sorted_vals):
        cumsum += v
        gini_sum += (2 * (i + 1) - n - 1) * v
    return gini_sum / (n * total)

def lorenz_curve(values, n_points=20):
    """Lorenz curve points for plotting."""
    sorted_vals = sorted(values)
    total = sum(sorted_vals)
    if total == 0:
        return [(0,0), (1,1)]
    n = len(sorted_vals)
    points = [(0, 0)]
    cumsum = 0
    for i, v in enumerate(sorted_vals):
        cumsum += v
        if (i+1) % max(1, n // n_points) == 0 or i == n-1:
            points.append(((i+1)/n, cumsum/total))
    return points

def theil_index(values):
    """Theil T index (GE(1)) for inequality."""
    vals = [v for v in values if v > 0]
    if not vals:
        return 0.0
    n = len(vals)
    mu = sum(vals) / n
    if mu == 0:
        return 0.0
    return sum((v/mu) * math.log(v/mu) for v in vals) / n

def atkinson_index(values, epsilon=1.0):
    """Atkinson index with inequality aversion parameter."""
    vals = [v for v in values if v > 0]
    if not vals:
        return 0.0
    n = len(vals)
    mu = sum(vals) / n
    if mu == 0:
        return 0.0
    if epsilon == 1.0:
        geo_mean = math.exp(sum(math.log(v) for v in vals) / n)
        return 1 - geo_mean / mu
    else:
        power_mean = (sum((v/mu)**(1-epsilon) for v in vals) / n) ** (1/(1-epsilon))
        return 1 - power_mean

def shannon_entropy(values):
    """Shannon entropy in bits."""
    total = sum(values)
    if total == 0:
        return 0.0
    probs = [v/total for v in values if v > 0]
    return -sum(p * math.log2(p) for p in probs)

def normalized_entropy(values):
    """Normalized Shannon entropy (0-1)."""
    h = shannon_entropy(values)
    n_nonzero = sum(1 for v in values if v > 0)
    h_max = math.log2(n_nonzero) if n_nonzero > 1 else 1
    return h / h_max if h_max > 0 else 0

def kl_divergence(p_values, q_values):
    """KL divergence D(P||Q) in bits."""
    p_total = sum(p_values)
    q_total = sum(q_values)
    if p_total == 0 or q_total == 0:
        return 0.0
    p_probs = [v/p_total for v in p_values]
    q_probs = [v/q_total for v in q_values]
    kl = 0
    for p, q in zip(p_probs, q_probs):
        if p > 0 and q > 0:
            kl += p * math.log2(p / q)
    return kl

def hhi_index(values):
    """Herfindahl-Hirschman Index (0-10000)."""
    total = sum(values)
    if total == 0:
        return 0
    shares = [(v/total)*100 for v in values]
    return sum(s**2 for s in shares)

def rate_ratio(count_a, pop_a, count_b, pop_b):
    """Rate ratio with 95% CI (Wald)."""
    rate_a = count_a / pop_a if pop_a > 0 else 0
    rate_b = count_b / pop_b if pop_b > 0 else 0
    rr = rate_a / rate_b if rate_b > 0 else float('inf')
    if count_a > 0 and count_b > 0:
        se_log_rr = math.sqrt(1/count_a + 1/count_b)
        ci_lo = math.exp(math.log(rr) - 1.96 * se_log_rr)
        ci_hi = math.exp(math.log(rr) + 1.96 * se_log_rr)
    else:
        ci_lo, ci_hi = 0, float('inf')
    return {"rr": round(rr, 3), "ci_lower": round(ci_lo, 3), "ci_upper": round(ci_hi, 3)}

def odds_ratio(a, b, c, d):
    """Odds ratio from 2x2 table with CI."""
    if b == 0 or c == 0:
        return {"or": float('inf'), "ci_lower": 0, "ci_upper": float('inf')}
    or_val = (a * d) / (b * c)
    se = math.sqrt(1/max(a,1) + 1/max(b,1) + 1/max(c,1) + 1/max(d,1))
    ci_lo = math.exp(math.log(or_val) - 1.96 * se) if or_val > 0 else 0
    ci_hi = math.exp(math.log(or_val) + 1.96 * se) if or_val > 0 else float('inf')
    return {"or": round(or_val, 3), "ci_lower": round(ci_lo, 3), "ci_upper": round(ci_hi, 3)}

def chi_squared_test(observed, expected):
    """Chi-squared test statistic and approximate p-value."""
    if len(observed) != len(expected):
        return {"chi2": 0, "df": 0, "p": 1}
    chi2 = sum((o-e)**2/e for o, e in zip(observed, expected) if e > 0)
    df = len(observed) - 1
    # Wilson-Hilferty approximation for p-value
    if df > 0:
        z = ((chi2/df)**(1/3) - (1-2/(9*df))) / math.sqrt(2/(9*df))
        p = 0.5 * math.erfc(z / math.sqrt(2))
    else:
        p = 1.0
    return {"chi2": round(chi2, 2), "df": df, "p": round(p, 4)}

def permutation_test(group_a, group_b, stat_fn=None, n_perm=10000, seed=42):
    """Two-sample permutation test."""
    rng = random.Random(seed)
    if stat_fn is None:
        stat_fn = lambda x: sum(x)/len(x) if x else 0
    observed_diff = stat_fn(group_a) - stat_fn(group_b)
    combined = group_a + group_b
    n_a = len(group_a)
    count_extreme = 0
    for _ in range(n_perm):
        rng.shuffle(combined)
        perm_diff = stat_fn(combined[:n_a]) - stat_fn(combined[n_a:])
        if abs(perm_diff) >= abs(observed_diff):
            count_extreme += 1
    return {"observed_diff": round(observed_diff, 4), "p": round(count_extreme/n_perm, 4)}

def poisson_rate(count, exposure, ci=True):
    """Poisson rate with exact CI (Garwood)."""
    rate = count / exposure if exposure > 0 else 0
    if ci and count > 0:
        # Garwood approximation
        lo = count * (1 - 1/(9*count) - 1.96/(3*math.sqrt(count)))**3 / exposure
        hi = (count+1) * (1 - 1/(9*(count+1)) + 1.96/(3*math.sqrt(count+1)))**3 / exposure
        return {"rate": round(rate, 4), "ci_lower": round(max(0, lo), 4), "ci_upper": round(hi, 4)}
    return {"rate": round(rate, 4)}

def morans_i(values, weights_matrix):
    """Moran's I spatial autocorrelation."""
    n = len(values)
    if n < 2:
        return {"I": 0, "expected": 0, "z": 0}
    mean_val = sum(values) / n
    numerator = 0
    denominator = sum((v - mean_val)**2 for v in values)
    total_w = 0
    for i in range(n):
        for j in range(n):
            w = weights_matrix[i][j] if i < len(weights_matrix) and j < len(weights_matrix[i]) else 0
            numerator += w * (values[i] - mean_val) * (values[j] - mean_val)
            total_w += w
    if denominator == 0 or total_w == 0:
        return {"I": 0, "expected": -1/(n-1), "z": 0}
    I = (n / total_w) * (numerator / denominator)
    expected = -1 / (n - 1)
    return {"I": round(I, 4), "expected": round(expected, 4)}

def bayesian_rate(successes, trials, prior_alpha=1, prior_beta=1):
    """Bayesian rate estimation (Beta-Binomial conjugate)."""
    post_alpha = prior_alpha + successes
    post_beta = prior_beta + (trials - successes)
    mean = post_alpha / (post_alpha + post_beta)
    # 95% credible interval using normal approximation to Beta
    var = (post_alpha * post_beta) / ((post_alpha + post_beta)**2 * (post_alpha + post_beta + 1))
    sd = math.sqrt(var)
    return {
        "mean": round(mean, 4),
        "ci_lower": round(max(0, mean - 1.96*sd), 4),
        "ci_upper": round(min(1, mean + 1.96*sd), 4),
        "posterior_alpha": post_alpha,
        "posterior_beta": post_beta
    }

def interrupted_time_series(pre_values, post_values):
    """Interrupted time series — level change + slope change."""
    n_pre = len(pre_values)
    n_post = len(post_values)
    if n_pre < 2 or n_post < 2:
        return {"level_change": 0, "slope_change": 0}
    # Pre-trend
    pre_slope = (pre_values[-1] - pre_values[0]) / (n_pre - 1) if n_pre > 1 else 0
    pre_intercept = pre_values[0]
    # Expected post value (extrapolation)
    expected_post_start = pre_intercept + pre_slope * n_pre
    # Actual
    level_change = post_values[0] - expected_post_start
    post_slope = (post_values[-1] - post_values[0]) / (n_post - 1) if n_post > 1 else 0
    slope_change = post_slope - pre_slope
    return {
        "level_change": round(level_change, 2),
        "slope_change": round(slope_change, 4),
        "pre_slope": round(pre_slope, 4),
        "post_slope": round(post_slope, 4)
    }

def changepoint_detection(values, min_segment=3):
    """Simple changepoint detection (max cumulative sum deviation)."""
    n = len(values)
    if n < 2 * min_segment:
        return {"changepoint": n // 2, "score": 0}
    mean_all = sum(values) / n
    best_score = 0
    best_idx = n // 2
    cusum = 0
    for i in range(min_segment, n - min_segment):
        cusum += values[i] - mean_all
        score = abs(cusum)
        if score > best_score:
            best_score = score
            best_idx = i
    return {"changepoint": best_idx, "score": round(best_score, 2)}

def power_law_fit(values):
    """MLE power-law exponent + KS goodness-of-fit."""
    vals = sorted([v for v in values if v > 0], reverse=True)
    if len(vals) < 5:
        return {"alpha": 0, "xmin": 0, "ks": 1}
    xmin = vals[-1]
    n = len(vals)
    alpha = 1 + n / sum(math.log(v / xmin) for v in vals if v > xmin)
    # KS statistic
    ks_max = 0
    for i, v in enumerate(vals):
        empirical = (i + 1) / n
        theoretical = 1 - (xmin / v) ** (alpha - 1) if v >= xmin else 0
        ks = abs(empirical - theoretical)
        if ks > ks_max:
            ks_max = ks
    return {"alpha": round(alpha, 3), "xmin": xmin, "ks": round(ks_max, 4)}

def concentration_index(values):
    """Concentration index (related to Gini, health-specific)."""
    return gini_coefficient(values)

def kaplan_meier(event_times, censored=None):
    """Kaplan-Meier survival estimates."""
    if censored is None:
        censored = [False] * len(event_times)
    events = sorted(zip(event_times, censored), key=lambda x: x[0])
    n_at_risk = len(events)
    survival = 1.0
    curve = [(0, 1.0)]
    for time, is_censored in events:
        if not is_censored:
            survival *= (n_at_risk - 1) / n_at_risk
            curve.append((time, survival))
        n_at_risk -= 1
    return curve

def spearman_correlation(x, y):
    """Spearman rank correlation."""
    n = len(x)
    if n < 3:
        return {"rho": 0, "p": 1}
    rank_x = _rank(x)
    rank_y = _rank(y)
    d_sq = sum((rx - ry)**2 for rx, ry in zip(rank_x, rank_y))
    rho = 1 - 6 * d_sq / (n * (n**2 - 1))
    # Approximate p-value
    t = rho * math.sqrt((n-2)/(1-rho**2)) if abs(rho) < 1 else 0
    return {"rho": round(rho, 4), "t": round(t, 3)}

def _rank(values):
    """Compute ranks for Spearman."""
    indexed = sorted(enumerate(values), key=lambda x: x[1])
    ranks = [0] * len(values)
    for rank, (idx, _) in enumerate(indexed):
        ranks[idx] = rank + 1
    return ranks

def linear_regression(x, y):
    """Simple linear regression."""
    n = len(x)
    if n < 2:
        return {"slope": 0, "intercept": 0, "r_squared": 0}
    mean_x = sum(x) / n
    mean_y = sum(y) / n
    ss_xy = sum((xi - mean_x) * (yi - mean_y) for xi, yi in zip(x, y))
    ss_xx = sum((xi - mean_x)**2 for xi in x)
    ss_yy = sum((yi - mean_y)**2 for yi in y)
    slope = ss_xy / ss_xx if ss_xx > 0 else 0
    intercept = mean_y - slope * mean_x
    r_squared = (ss_xy**2) / (ss_xx * ss_yy) if ss_xx > 0 and ss_yy > 0 else 0
    return {"slope": round(slope, 4), "intercept": round(intercept, 4), "r_squared": round(r_squared, 4)}

def effect_size_cohens_d(mean1, sd1, n1, mean2, sd2, n2):
    """Cohen's d with pooled SD."""
    pooled_sd = math.sqrt(((n1-1)*sd1**2 + (n2-1)*sd2**2) / (n1+n2-2))
    if pooled_sd == 0:
        return {"d": 0, "ci_lower": 0, "ci_upper": 0}
    d = (mean1 - mean2) / pooled_sd
    se = math.sqrt((n1+n2)/(n1*n2) + d**2/(2*(n1+n2)))
    return {"d": round(d, 3), "ci_lower": round(d - 1.96*se, 3), "ci_upper": round(d + 1.96*se, 3)}

def network_centrality(adjacency):
    """Degree centrality from adjacency matrix."""
    n = len(adjacency)
    if n < 2:
        return []
    centralities = []
    for i in range(n):
        degree = sum(1 for j in range(n) if adjacency[i][j] > 0 and i != j)
        centralities.append(degree / (n - 1))
    return centralities

def jaccard_similarity(set_a, set_b):
    """Jaccard similarity coefficient."""
    intersection = len(set_a & set_b)
    union = len(set_a | set_b)
    return intersection / union if union > 0 else 0

# lib/data_fetcher.py
"""Fetch trial data from ClinicalTrials.gov API v2 for each paper topic."""
import json, requests, time, os
from pathlib import Path

API_BASE = "https://clinicaltrials.gov/api/v2/studies"
CACHE_DIR = Path("C:/Users/user/africa-e156-students/data_cache")

def fetch_trial_count(condition=None, location=None, other_terms=None):
    """Fetch total trial count for a query."""
    params = {"format": "json", "pageSize": 1, "countTotal": "true",
              "filter.advanced": "AREA[StudyType]INTERVENTIONAL"}
    if condition:
        params["query.cond"] = condition
    if location:
        params["query.locn"] = location
    if other_terms:
        params["query.term"] = other_terms
    try:
        resp = requests.get(API_BASE, params=params, timeout=30)
        return resp.json().get("totalCount", 0)
    except:
        return 0

def fetch_studies(condition=None, location=None, other_terms=None, max_results=200):
    """Fetch study records for deeper analysis."""
    params = {"format": "json", "pageSize": min(max_results, 200),
              "filter.advanced": "AREA[StudyType]INTERVENTIONAL"}
    if condition:
        params["query.cond"] = condition
    if location:
        params["query.locn"] = location
    if other_terms:
        params["query.term"] = other_terms
    try:
        resp = requests.get(API_BASE, params=params, timeout=60)
        return resp.json().get("studies", [])
    except:
        return []

def fetch_paper_data(paper_def):
    """Fetch all data needed for a single paper. Uses cache."""
    CACHE_DIR.mkdir(exist_ok=True)
    cache_file = CACHE_DIR / f"{paper_def['slug']}.json"
    if cache_file.exists():
        with open(cache_file, encoding="utf-8") as f:
            return json.load(f)

    q = paper_def["query"]
    data = {}
    # Africa count
    data["africa_count"] = fetch_trial_count(
        condition=q.get("condition"), location="Africa",
        other_terms=q.get("other"))
    time.sleep(0.3)
    # US count
    data["us_count"] = fetch_trial_count(
        condition=q.get("condition"), location="United States",
        other_terms=q.get("other"))
    time.sleep(0.3)
    # Europe count
    data["europe_count"] = fetch_trial_count(
        condition=q.get("condition"), location="Europe",
        other_terms=q.get("other"))
    time.sleep(0.3)
    # Africa studies (for deeper analysis)
    data["africa_studies"] = fetch_studies(
        condition=q.get("condition"), location=q.get("location", "Africa"),
        other_terms=q.get("other"), max_results=200)
    time.sleep(0.3)
    # Country-specific if needed
    if q.get("countries"):
        data["country_counts"] = {}
        for country in q["countries"]:
            data["country_counts"][country] = fetch_trial_count(
                condition=q.get("condition"), location=country,
                other_terms=q.get("other"))
            time.sleep(0.2)

    # Cache
    with open(cache_file, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, default=str)
    return data

### Task 9: Main Orchestrator

**Files:**
- Create: `expand_to_190.py`

The master script that imports all modules and runs the full pipeline.

- [ ] **Step 1: Implement orchestrator**

```python
#!/usr/bin/env python3
"""
Expand Africa E156 Student Platform from 80 to 190 papers.
Generates 110 new papers with E156 bodies, Python scripts, and HTML dashboards.
"""
import sys, io, json, time
from pathlib import Path
from lib.paper_manifest import MANIFEST
from lib.data_fetcher import fetch_paper_data
from lib.stats_library import *
from lib.body_generator import generate_body
from lib.chart_library import *
from lib.dashboard_generator import generate_dashboard
from lib.code_generator import generate_code_script
from lib.index_updater import update_group_index, update_root_index

sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8', errors='replace')
OUT = Path("C:/Users/user/africa-e156-students")

def main():
    print(f"=== Expanding to 190 papers ({len(MANIFEST)} new) ===")
    results = {"pass": 0, "fail": 0, "errors": []}

    for i, paper in enumerate(MANIFEST):
        slug = paper["slug"]
        group = paper["group"]
        print(f"\n[{i+1}/{len(MANIFEST)}] {slug}")

        try:
            # 1. Fetch data
            data = fetch_paper_data(paper)
            print(f"  Data: AF={data.get('africa_count',0)}, US={data.get('us_count',0)}")

            # 2. Compute statistics
            stats_results = compute_paper_stats(paper, data)

            # 3. Generate E156 body
            body = generate_body(paper, data, stats_results)
            words = len(body.split())
            sentences = body.count('.') + body.count('?')
            print(f"  Body: {words} words, {sentences} sentences")

            # 4. Write Python script
            code_path = OUT / group / "code" / f"{slug.replace('_','-')}.py"
            code_path.parent.mkdir(parents=True, exist_ok=True)
            code = generate_code_script(paper, data, stats_results)
            code_path.write_text(code, encoding="utf-8")

            # 5. Write HTML dashboard
            dash_path = OUT / group / "dashboards" / f"{slug}.html"
            dash_path.parent.mkdir(parents=True, exist_ok=True)
            dashboard = generate_dashboard(paper, data, stats_results, body)
            dash_path.write_text(dashboard, encoding="utf-8")

            # 6. Validate
            if words > 170 or sentences != 7:
                results["errors"].append(f"{slug}: {words}w, {sentences}s")
            results["pass"] += 1

        except Exception as e:
            print(f"  ERROR: {e}")
            results["fail"] += 1
            results["errors"].append(f"{slug}: {e}")

    # 7. Update index pages
    print("\n=== Updating index pages ===")
    for group in ["geographic-equity", "health-disease", "governance-justice", "methods-systems"]:
        group_papers = [p for p in MANIFEST if p["group"] == group]
        update_group_index(OUT / group / "index.html", group_papers)
    update_root_index(OUT / "index.html")

    # 8. Report
    print(f"\n=== DONE ===")
    print(f"Pass: {results['pass']}, Fail: {results['fail']}")
    if results["errors"]:
        print("Errors:")
        for e in results["errors"]:
            print(f"  - {e}")

    with open(OUT / "expansion_report.json", "w") as f:
        json.dump(results, f, indent=2)

if __name__ == "__main__":
    main()
```
